fn main() {
    let i: i32 = 5;
    let b: bool = false;

    // Logical‐operand errors
    let l1 = i || b;   // ✘ '||' requires both operands to be bool
    let l2 = !i;       // ✘ '!' requires a bool operand

    // Arithmetic‐operand errors
    let a1 = b + 10;   // ✘ '+' requires both operands to be i32
    let a2 = b * i;    // ✘ '*' requires both operands to be i32

    println!(
        "l1 = {l1}, l2 = {l2}, a1 = {a1}, a2 = {a2}",
        l1 = l1, l2 = l2, a1 = a1, a2 = a2
    );
}
