// Function returns a bool
fn is_positive(n: i32) -> bool {
    return n > 0;
}

fn main() {
    // x is declared as an i32
    let mut x: i32;

    // ❌ Semantic Rule 10 violated:
    // The right-hand side is a bool (from is_positive),
    // but x is of type i32
    x = is_positive(5);

    println!("x = {x}", x=x);
}
