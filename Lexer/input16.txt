// Declared to return i32, but returns a bool → semantic error!
fn get_flag() -> i32 {
    return false;
}

fn main() {
    // Even if you ignore get_flag’s body, assigning its result to an i32
    // would also violate Rule 10—but the primary failure is the return in get_flag.
    let mut x: i32 = 42;
    x = get_flag();
    println!("x = {x}", x=x);
}
